\section*{Basic String Paths}

\subsection*{Upward Cup (s style)}

\begin{center}
\tz{
  \draw[ultra thick, strA] (0,0) to[s] (1,2);
  \draw[ultra thick, strB] (2,0) to[s] (1,2);
  \mk{(0,0)};
  \mk{(2,0)};
  \mk{(1,2)};
}
\end{center}

\subsection*{Downward Cap (-s style)}

\begin{center}
\tz{
  \draw[ultra thick, strA] (1,0) to[-s] (0,2);
  \draw[ultra thick, strB] (1,0) to[-s] (2,2);
  \mk{(1,0)};
  \mk{(0,2)};
  \mk{(2,2)};
}
\end{center}

\subsection*{Identity Strands}

\begin{center}
\tz{
  \draw[ultra thick, strA] (0,0) -- (0,2);
  \draw[ultra thick, strB] (1,0) -- (1,2);
  \draw[ultra thick, strC] (2,0) -- (2,2);
  \node[below] at (1,-0.3) {Three parallel strands};
}
\end{center}

\section*{String Diagram Compositions}

\subsection*{Cup-Cap Pair}

\begin{center}
\tz[scale=1.2]{
  % Cup at bottom
  \draw[ultra thick, strA] (0,0) to[s] (1,1);
  \draw[ultra thick, strA] (2,0) to[s] (1,1);
  % Cap at top
  \draw[ultra thick, strB] (1,1) to[-s] (0,2);
  \draw[ultra thick, strB] (1,1) to[-s] (2,2);
}
\end{center}

\subsection*{Braiding}

\begin{center}
\tz[scale=1.3]{
  % Left strand goes over
  \draw[ultra thick, strA] (0,0) to[s] (1,1) to[-s] (2,2);
  % Right strand goes under (with knot)
  \draw[ultra thick, strB, knot] (2,0) to[s] (1,1) to[-s] (0,2);
  
  \mk[strA]{(0,0)};
  \mk[strA]{(2,2)};
  \mk[strB]{(2,0)};
  \mk[strB]{(0,2)};
}
\end{center}

\subsection*{Tensor Product}

\begin{center}
\tz{
  % Two separate string diagrams side by side
  \begin{scope}
    \draw[ultra thick, strA] (0,0) to[s] (0.5,1) to[-s] (1,2);
    \draw[ultra thick, strB] (1,0) to[s] (0.5,1) to[-s] (0,2);
  \end{scope}
  
  \node at (2,1) {$\otimes$};
  
  \begin{scope}[shift={(3,0)}]
    \draw[ultra thick, strC] (0,0) -- (0,2);
    \draw[ultra thick, strD] (1,0) -- (1,2);
  \end{scope}
}
\end{center}

\subsection*{Yanking}

\begin{center}
\tz[scale=1.5]{
  % Yanking diagram: cup then cap
  \draw[ultra thick, strA] (0,0) to[s] (0.75,1.5);
  \draw[ultra thick, strA] (1.5,0) to[s] (0.75,1.5);
  \draw[ultra thick, strA] (0.75,1.5) to[-s] (0,3);
  \draw[ultra thick, strA] (0.75,1.5) to[-s] (1.5,3);
  
  \node at (2.5,1.5) {$\cong$};
  
  \begin{scope}[shift={(3.5,0)}]
    \draw[ultra thick, strA] (0,0) -- (0,3);
    \draw[ultra thick, strA] (0.75,0) -- (0.75,3);
  \end{scope}
}
\end{center}
