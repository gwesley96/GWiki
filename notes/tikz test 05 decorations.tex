\section*{Physics Applications}

\subsection*{Spring and Wave}

\begin{center}
\tz[scale=0.9]{
  % Mass-spring system
  \draw[thick] (0,0) -- (0,2);
  \draw[thick, coil] (0,2) -- (0,3.5);
  \fill[gray] (-0.3,3.5) rectangle (0.3,4);

  % Wave
  \draw[ultra thick, blue, squiggly] (1.5,2) -- (4.5,2);
  \draw[->, thick] (4.7,2) -- (5.2,2);

  \node at (0.15,2.75) [right] {spring};
  \node at (3,2.5) {wave};
}
\end{center}

\subsection*{Feynman-style Diagram}

\begin{center}
\tz{
  % Particles
  \draw[thick, ->] (0,1) -- (1.5,1);
  \draw[thick, ->] (0,-1) -- (1.5,-1);

  % Interaction
  \draw[thick, snake, red] (1.5,1) -- (1.5,-1);

  % Outgoing
  \draw[thick, ->] (1.5,1) -- (3,0.5);
  \draw[thick, ->] (1.5,-1) -- (3,-0.5);

  \node at (0.75,1.3) {$e^-$};
  \node at (0.75,-1.3) {$e^+$};
  \node at (2,0) [right] {$\gamma$};
}
\end{center}
