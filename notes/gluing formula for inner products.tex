\begin{lemma}[Gluing formula for inner products]
\label{lem:codim-1 gluing formula}
Let $(\mathcal{F}, \mathcal{U}, z)$ be an $(n{+}1)$-dimensional unitary TQFT. Suppose $X \in \mathcal{M}_n$ has boundary
\[
\partial X = (\overline{Y} \sqcup Y)_{\overline{\partial Y} \sqcup \partial Y} \cup Q
\]
for $Y, Q \in \mathcal{M}_{n-1}$. Fix $d \in \mathcal{F}(\partial Y)$, $q \in \mathcal{F}(Q; \overline{d} \sqcup d)$, and $y \in \mathcal{F}(Y, d)$. Then for $f, f' \in \mathcal{A}(X, (\overline{y} \sqcup y) \bullet q)$,
\[
\langle f_\glu, f'_\glu \rangle_{X_\glu, q_\glu}
= \sum_{s \in \mathrm{Irr}} \sum_{k=1}^{n_s}
\frac{\langle f \cup e_s^{(k)}, f' \cup e_s^{(k)} \rangle_{X, (\overline{y} \sqcup y) \bullet q}}{\langle e_s^{(k)}, e_s^{(k)} \rangle_{Y \times I, \overline{y} \sqcup y}},
\]
where $\mathrm{Irr} = \mathrm{Irr}(\mathsf{Rep}(\mathrm{End}_{\mathcal{A}(Y,d)}(y)))$, and $\{e_s^{(k)}\}$ is a complete set of orthogonal minimal idempotents for the unitary algebra $\mathrm{End}_{\mathcal{A}(Y,d)}(y)$.
\end{lemma}


The next result shows how gluing $n$-manifolds affects the TQFT inner products.

\begin{proof}
Since $\End_{\A(Y,d)}(y)$ is a finite-dimensional unitary algebra, it admits a complete set of central, minimal, orthogonal projections $e_s^{(k)}\colon y\to y$, indexed over $k=1,\dots,n_s$ and over simples $s\in\Irr(\Rep\End_{\A(Y,d)}(y))$. \greyson{The first sentence in the following proof uses that (i) skein spaces of $n$-manifolds are finite-dimensional vector spaces and (ii) that cylinder 1-categories are unitary (to get that the endomorphism spaces are unitary). However, the latter is proven in the proof of a lemma that itself uses finiteness; thus, to avoid circularity, \cref{lem:codim-1 gluing formula} and its proof (the proof you are reading right now) should go AFTER \cref{f.d. spaces}, AFTER the proof of \emph{unitarity} in \cref{unitary implies finite unitary cylinder 1-categories}, but BEFORE the finiteness proof of \cref{unitary implies finite unitary cylinder 1-categories}!}

Thus $e_s^{(k)}\circ e_t^{(\ell)}=\delta_{st}\delta_{k\ell}e_s^{(k)}$, so
\[
    \bkt{e_s^{(k)}}{e_t^{(\ell)}}_{Y\times I,\fcj y\sqcup y}
    =
    z_{Y\times I\times I}(\fcj{e_{s}^{(k)}}\bullet_{\fcj y\sqcup y} e_t^{(\ell)})
    =
    \delta_{st}\delta_{k\ell}\bkt{e_s^{(k)}}{e_s^{(k)}},
\]
which is strictly positive when $s=t$ and $k=\ell$ by positive-definiteness (ensuring $e_s^{(k)}\neq 0$), and zero otherwise.

To verify spanning, note that by the Artin-Wedderburn theorem, $\End_{\A(Y,d)}(y) \cong \bigoplus_s M_{n_s}(\mathbb{C})$, and the projections satisfy the resolution of identity $\sum_{s,k} e_s^{(k)} = \id_y$. For any $f \in \A(Y\times I,\fcj y\sqcup y)$, we have
\[
f = \id_y \circ f \circ \id_y = \sum_{s,k,t,\ell} e_s^{(k)} \circ f \circ e_t^{(\ell)} \in \mathrm{span}\{e_s^{(k)}\}_{s,k},
\]
since each summand is a scalar multiple of some $e_r^{(m)}$ by the orthogonality relations. Thus $\{e_s^{(k)}\}_{s,k}$ is an orthogonal basis of $\A(Y\times I,\fcj y\sqcup y)$.

Applying the gluing formula with corners for the gluing $X\times I\to (X\times I)_\glu=X_\glu\times I$, we find
\[
\bkt{f_\glu}{f_\glu'}=z_{X_\glu\times I}(\fcj {f_\glu}\bullet_{q_\glu}f_\glu')=\sum_{s,k}\frac{z_{X\times I}((\fcj{f}\cup\fcj{e_s^{(k)}})\bullet (f'\cup e_s^{(k)}))}{\bkt{e_s^{(k)}}{e_s^{(k)}}_{Y\times I,\fcj y\sqcup y}}.
\]
The idempotents decompose $\A(X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d}q)$ into matrix blocks. By inserting the resolution of identity $\id_y = \sum_{s,k} e_s^{(k)}$ on both boundaries, any $f$ can be written as
\[
f = \sum_{s,k,t,\ell} e_s^{(k)}\cup f\cup e_t^{(\ell)},
\]
where each component lives in the space $e_s^{(k)}\cup \A(X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d}q)\cup e_t^{(\ell)}$. Since $e_s^{(k)}\circ e_t^{(\ell)}=\delta_{st}\delta_{k\ell}e_s^{(k)}$,
\[
(e_s^{(k)}\cup f\cup e_t^{(\ell)})\cup e_{r}^{(m)} = e_s^{(k)}\cup f\cup (e_t^{(\ell)}\circ e_{r}^{(m)}) = \delta_{tr}\delta_{\ell m}\cdot e_s^{(k)}\cup f\cup e_t^{(\ell)}
\]
and similarly $\fcj{e_{r}^{(m)}}\cup (e_s^{(k)}\cup f\cup e_t^{(\ell)}) = \delta_{sr}\delta_{km}\cdot e_s^{(k)}\cup f\cup e_t^{(\ell)}$. Thus
\[
f\cup e_{r}^{(m)} = \sum_{s,k,t,\ell} \delta_{tr}\delta_{\ell m}\cdot e_s^{(k)}\cup f\cup e_t^{(\ell)} = \sum_{s,k} e_s^{(k)}\cup f\cup e_{r}^{(m)}.
\]
For each corner $(r,m)=(s,k)$, choose an orthogonal basis $\{f_{s,k,i}\}_i$ of $e_s^{(k)}\cup \A(X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d}q)\cup e_s^{(k)}$. The entire collection $\{f_{s,k,i}\}_{s,k,i}$ forms an orthogonal basis of $\A(X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d}q)$, and by the above, $f_{s,k,i}\cup e_{t}^{(\ell)}=\delta_{st}\delta_{k\ell}f_{s,k,i}$ and $\fcj{e_t^{(\ell)}}\cup f_{s,k,i}=\delta_{st}\delta_{k\ell}f_{s,k,i}$. For these basis elements,
\begin{align*}
     \bkt{\glu(f_{s,k,i})}{\glu(f_{t,\ell,j})}
     &=\sum_{r,m}\frac{\bkt{f_{s,k,i}\cup e_{r}^{(m)}}{f_{t,\ell,j}\cup e_{r}^{(m)}}_{X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d} q}}{\bkt{e_{r}^{(m)}}{e_{r}^{(m)}}_{Y\times I,\fcj y\sqcup y}}\\
     &=\sum_{r,m}\frac{\delta_{sr}\delta_{km}\delta_{tr}\delta_{\ell m}\bkt{f_{s,k,i}}{f_{t,\ell,j}}_{X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d} q}}{\bkt{e_{r}^{(m)}}{e_{r}^{(m)}}_{Y\times I,\fcj y\sqcup y}}\\
     &=
     \delta_{st}\delta_{k\ell}\delta_{ij}\frac{\bkt{f_{s,k,i}}{f_{s,k,i}}_{X,(\fcj y\sqcup y)\bullet_{\fcj d\sqcup d} q}}{\bkt{e_{s}^{(k)}}{e_{s}^{(k)}}_{Y\times I,\fcj y\sqcup y}}.
 \end{align*}
Thus, for general $f,f'$ as in the claim, expanding in this orthogonal basis and using bilinearity of the inner product, we obtain
\[
\bkt{f_\glu}{f'_\glu}_{X_\glu,q_\glu}
=
\sum_{s,k}\frac{\bkt{f\cup e_s^{(k)}}{f'\cup e_s^{(k)}}_{X, (\fcj{y}\sqcup y)\bullet_{\fcj d\sqcup d} q}}{\bkt{e_s^{(k)}}{e_s^{(k)}}_{Y \times I,\fcj y\sqcup y}}.\qedhere
\]
\end{proof}


\begin{example}
When $y$ is \emph{simple}, meaning $\mathrm{End}(y) \cong \mathbb{C}$, there is a unique minimal idempotent $e = \mathrm{id}_y$, and the formula becomes
\[
\bkt{f_\glu}{f'_\glu}_{X_\glu,q_\glu}
%  = \frac{\langle f, f' \rangle_X}{\langle \mathrm{id}_y, \mathrm{id}_y \rangle_{Y \times I}} 
 = \frac{\bkt{f}{f'}_{X, (\fcj{y}\sqcup y)\bullet_{\fcj d\sqcup d} q}}{d_y},
\]
where $d_y = \langle \mathrm{id}_y, \mathrm{id}_y \rangle_{Y\times I,\fcj y\sqcup y}$.
\end{example}
